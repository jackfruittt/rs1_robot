171  bool WaypointPlanner::segmentsIntersect(
172  const geometry_msgs::msg::Point& p1,
173  const geometry_msgs::msg::Point& p2,
174  const geometry_msgs::msg::Point& p3,
175  const geometry_msgs::msg::Point& p4) const {
176 
177  // Use cross product method to determine if line segments intersect
178  // Segment 1: p1 -> p2
179  // Segment 2: p3 -> p4
180 
181  auto sign = [](double val) { return (val > 0.0) ? 1 : ((val < 0.0) ? -1 : 0); };
182 
183  auto cross_product = [](double ax, double ay, double bx, double by) {
184  return ax * by - ay * bx;
185  };
186 
187  // Vector from p1 to p2
188  double d1x = p2.x - p1.x;
189  double d1y = p2.y - p1.y;
190 
191  // Vector from p3 to p4
192  double d2x = p4.x - p3.x;
193  double d2y = p4.y - p3.y;
194 
195  // Check if segments are parallel (cross product near zero)
196  double cross_d1_d2 = cross_product(d1x, d1y, d2x, d2y);
197  if (std::abs(cross_d1_d2) < 1e-10) {
198  return false; // Parallel or collinear, treat as non-intersecting
199  }
200 
201  // Vector from p1 to p3
202  double d3x = p3.x - p1.x;
203  double d3y = p3.y - p1.y;
204 
205  // Check if p3 and p4 are on opposite sides of line through p1-p2
206  double cross1 = cross_product(d1x, d1y, d3x, d3y);
207  double cross2 = cross_product(d1x, d1y, p4.x - p1.x, p4.y - p1.y);
208 
209  if (sign(cross1) == sign(cross2) && sign(cross1) != 0) {
210  return false; // Same side, no intersection
211  }
212 
213  // Check if p1 and p2 are on opposite sides of line through p3-p4
214  double cross3 = cross_product(d2x, d2y, -d3x, -d3y);
215  double cross4 = cross_product(d2x, d2y, p2.x - p3.x, p2.y - p3.y);
216 
217  if (sign(cross3) == sign(cross4) && sign(cross3) != 0) {
218  return false; // Same side, no intersection
219  }
220 
221  return true; // Segments intersect
222  }

230  }
