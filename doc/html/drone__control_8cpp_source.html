540  // Very close obstacles (<2m): STOP forward motion entirely, climb only
541  // Medium distance (2-5m): Slow to 5%, keep climbing
542  const double CRITICAL_DISTANCE = 2.0; // Stop forward motion if closer than 2m
543 
544  if (min_obstacle_distance < CRITICAL_DISTANCE) {
545  speed_multiplier = 0.0; // STOP all horizontal movement
546  } else {
547  speed_multiplier = OBSTACLE_SLOWDOWN_FACTOR; // Slow to 5%
548  }
549 
550  if (!emergency_climb_active_) {
551  // Just detected obstacle - start emergency climb and timer
552  clearance_target_altitude_ = sonar_range + emergency_climb_altitude_;
553  clearance_target_altitude_ = std::min(clearance_target_altitude_, MAX_EMERGENCY_ALTITUDE);
554  emergency_climb_active_ = true;
555  emergency_climb_start_ = std::chrono::steady_clock::now();
556  panic_climb_active_ = false;
557  clearance_hold_active_ = false; // Reset the post-clear climb flag
558 
559  if (speed_multiplier == 0.0) {
560  RCLCPP_WARN(logger_, "CRITICAL: Obstacle %.2fm ahead - STOPPING forward motion, CLIMBING to %.2fm (current: %.2fm)",
561  min_obstacle_distance, clearance_target_altitude_, sonar_range);
562  } else {
563  RCLCPP_WARN(logger_, "Obstacle detected %.2fm ahead - SLOWING to %.0f%% speed, climbing to %.2fm (current: %.2fm)",
564  min_obstacle_distance, speed_multiplier * 100.0, clearance_target_altitude_, sonar_range);
565  }
566  }
567 
568  // Check if climbing duration is excessive (tall obstacle) - enter PANIC CLIMB
569  auto climb_duration = std::chrono::steady_clock::now() - emergency_climb_start_;
570  double climb_seconds = std::chrono::duration<double>(climb_duration).count();
571 
572  if (climb_seconds > PANIC_CLIMB_TIMEOUT && !panic_climb_active_) {
573  panic_climb_active_ = true;
574  // Add 2m safety margin above current altitude for tall obstacles
575  clearance_target_altitude_ = sonar_range + 2.0; // Just 2m above current position
576  clearance_target_altitude_ = std::min(clearance_target_altitude_, MAX_EMERGENCY_ALTITUDE);
577  RCLCPP_ERROR(logger_, "PANIC CLIMB ACTIVATED! Tall obstacle detected (%.1fs climbing). Ascending FAST to %.2fm!",
578  climb_seconds, clearance_target_altitude_);
579  }
580 
581  target_altitude = clearance_target_altitude_;
582 
583  // Keep climbing if clearance altitude not reached yet
584  if (sonar_range < clearance_target_altitude_ - 0.5) {
585  RCLCPP_DEBUG(logger_, "Climbing to clearance: %.2fm / %.2fm (obstacle at %.2fm)",
586  sonar_range, clearance_target_altitude_, min_obstacle_distance);
587  }
588 
589  } else if (emergency_climb_active_ && min_obstacle_distance > obstacle_detection_distance_ * 1.2) {
590  // Obstacle just cleared - continue climbing extra before resuming normal operations
591  // Reduced threshold from 1.5 to 1.2 (6m instead of 7.5m) for faster recovery
592  if (!clearance_hold_active_) {
593  // First time clearing - add extra climb height
594  clearance_target_altitude_ = sonar_range + EXTRA_CLIMB_AFTER_CLEAR;
595  clearance_target_altitude_ = std::min(clearance_target_altitude_, MAX_EMERGENCY_ALTITUDE);
596  clearance_hold_active_ = true;
597  RCLCPP_INFO(logger_, "Obstacle cleared! Climbing extra %.2fm to %.2fm for safety - resuming forward speed",
598  EXTRA_CLIMB_AFTER_CLEAR, clearance_target_altitude_);
599  }
600 
601  // Check if the extra climb has been completed
602  if (sonar_range >= clearance_target_altitude_ - 0.5) {
603  // Extra climb complete - now return to normal
604  emergency_climb_active_ = false;
605  clearance_hold_active_ = false;
606  panic_climb_active_ = false;
607  target_altitude = terrain_following_altitude_;
608  speed_multiplier = 1.0;
609  RCLCPP_INFO(logger_, "Extra climb complete at %.2fm - continuing at full speed", sonar_range);
610  } else {
611  // Still doing extra climb - resume normal forward speed while climbing
612  target_altitude = clearance_target_altitude_;
613  speed_multiplier = 1.0; // Full forward speed during extra climb
614  RCLCPP_DEBUG(logger_, "Extra climb in progress: %.2fm / %.2fm (full forward speed)", sonar_range, clearance_target_altitude_);
615  }
616 
617  } else if (emergency_climb_active_) {
618  // Still in emergency climb zone - maintain clearance altitude and slow speed
619  target_altitude = clearance_target_altitude_;
620  speed_multiplier = OBSTACLE_SLOWDOWN_FACTOR;
621  }
622 
623  // Calculate altitude control based on sonar reading
624  double altitude_error = target_altitude - sonar_range;
625  double altitude_command = pid_z_->calculateWithWindupProtection(target_altitude, sonar_range, dt, 1.0);
626 
627  // Apply panic climb speed if activated (overrides PID for tall obstacles)
628  if (panic_climb_active_) {
629  altitude_command = PANIC_CLIMB_SPEED; // Force ridiculous climb rate
630  RCLCPP_DEBUG(logger_, "PANIC CLIMB: Forcing vertical speed to %.1f m/s (altitude: %.2fm / %.2fm)",
631  altitude_command, sonar_range, target_altitude);
632  } else {
633  // Conservative altitude velocity limits for safety during normal operation
634  altitude_command = std::clamp(altitude_command, -5.0, 5.0); // +-5m/s max vertical velocity
635  }
636 
637  // Combine horizontal and vertical commands
638  geometry_msgs::msg::Twist cmd_vel = horizontal_cmd;
639 
640  // Apply speed reduction if obstacle detected
641  cmd_vel.linear.x *= speed_multiplier;
642  cmd_vel.linear.y *= speed_multiplier;
643  cmd_vel.angular.z *= speed_multiplier; // Also slow rotation
644 
645  cmd_vel.linear.z = altitude_command;
646 
647  // Periodic debug output for altitude control monitoring
648  static int altitude_log_counter = 0;
649  if (altitude_log_counter % 50 == 0) { // Every 5 seconds at 10Hz
650  RCLCPP_DEBUG(logger_, "Altitude control: sonar=%.2fm target=%.2fm error=%.2fm cmd_z=%.2f%s",
651  sonar_range, target_altitude, altitude_error, altitude_command,
652  emergency_climb_active_ ? " [EMERGENCY CLIMB]" : "");
653  }
654  altitude_log_counter++;
655 
656  return cmd_vel;
657  }

666  double DroneControl::checkForObstaclesAhead(const sensor_msgs::msg::LaserScan& lidar_data) {
667  if (lidar_data.ranges.empty()) {
668  return std::numeric_limits<double>::infinity(); // No data = no obstacles
669  }
670 
671  // Calculate forward-facing LiDAR indices (0deg is forward)
672  int total_rays = static_cast<int>(lidar_data.ranges.size());
673  double angle_increment = lidar_data.angle_increment;
674  double angle_min = lidar_data.angle_min;
675 
676  // Check forward sector (30-deg from forward direction)
677  double forward_sector_half_angle = M_PI / 6.0; // 30 degrees
678 
679  double min_distance = std::numeric_limits<double>::infinity();
680  int valid_readings = 0;
681 
682  for (int i = 0; i < total_rays; ++i) {
683  double angle = angle_min + i * angle_increment;
684 
685  // Normalize angle to [-, ]
686  angle = normalizeAngle(angle);
687 
688  // Check if ray is in forward-facing sector
689  if (std::abs(angle) <= forward_sector_half_angle) {
690  double range = lidar_data.ranges[i];
691 
692  // Filter out invalid readings
693  if (range >= lidar_data.range_min && range <= lidar_data.range_max &&
694  !std::isinf(range) && !std::isnan(range)) {
695  min_distance = std::min(min_distance, static_cast<double>(range));
696  valid_readings++;
697  }
698  }
699  }
700 
701  // Return infinity if no valid readings in forward sector
702  if (valid_readings == 0) {
703  return std::numeric_limits<double>::infinity();
704  }
705 
706  return min_distance;
707  }
