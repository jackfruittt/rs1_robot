216 bool MissionExecutor::executeDebrisReaction(
217  const geometry_msgs::msg::Point& debris_location,
218  WaypointPlanner* path_planner) {

225  RCLCPP_INFO(logger_, "=== DEBRIS INSPECTION MISSION STARTED ===");
226  RCLCPP_INFO(logger_, " Debris location: [%.2f, %.2f, %.2f]",
227  debris_location.x, debris_location.y, debris_location.z);
228 
229  // Generate circular orbit waypoints for debris inspection
230  // 5m radius orbit with 8 waypoints (45Â° intervals) for comprehensive view
231  const double ORBIT_RADIUS = 5.0; // meters
232  const int ORBIT_POINTS = 8; // waypoints around circle
233 
234  std::vector<geometry_msgs::msg::PoseStamped> orbit_waypoints;
235  orbit_waypoints.reserve(ORBIT_POINTS);
236 
237  constexpr double PI = 3.14159265358979323846;
238  const double angle_step = 2.0 * PI / static_cast<double>(ORBIT_POINTS);
239 
240  for (int i = 0; i < ORBIT_POINTS; ++i) {
241  const double theta = angle_step * static_cast<double>(i);
242 
243  // Calculate position on circle around debris
244  const double px = debris_location.x + ORBIT_RADIUS * std::cos(theta);
245  const double py = debris_location.y + ORBIT_RADIUS * std::sin(theta);
246 
247  // Calculate yaw to face the debris (centre point)
248  double yaw = std::atan2(debris_location.y - py, debris_location.x - px);
249 
250  // Convert yaw to quaternion (rotation around Z axis)
251  const double half_yaw = yaw * 0.5;
252  const double qw = std::cos(half_yaw);
253  const double qz = std::sin(half_yaw);
254 
255  // Create waypoint
256  geometry_msgs::msg::PoseStamped waypoint;
257  waypoint.header.frame_id = "map";
258  waypoint.header.stamp = rclcpp::Time(0);
259  waypoint.pose.position.x = px;
260  waypoint.pose.position.y = py;
261  waypoint.pose.position.z = debris_location.z; // Same altitude as debris
262  waypoint.pose.orientation.w = qw;
263  waypoint.pose.orientation.x = 0.0;
264  waypoint.pose.orientation.y = 0.0;
265  waypoint.pose.orientation.z = qz;
266 
267  orbit_waypoints.push_back(waypoint);
268  }
269 
270  // Set the orbit waypoints for execution
271  path_planner->setWaypoints(orbit_waypoints);
272 
273  RCLCPP_INFO(logger_, "Debris inspection waypoints configured:");
274  RCLCPP_INFO(logger_, " Orbit radius: %.1fm with %d waypoints", ORBIT_RADIUS, ORBIT_POINTS);
275  RCLCPP_INFO(logger_, " Drone will orbit debris for visual inspection");
